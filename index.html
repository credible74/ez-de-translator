// ------------- CONFIG & STORAGE ------------- let translationExamples = []; // Hier landen deine Sätze aus dem Sheet const CSV_URL = "https://docs.google.com/spreadsheets/d/1haD4IVA8lNqoUsnrMrnEHhf5iJIGVXGfTWWu5XxevJY/export?format=tsv"; // API-Konfiguration (Hier trägst du deinen API-Schlüssel ein) const OPENAI_API_KEY = "DEIN_OPENAI_API_KEY"; // ------------- DATA LOADER ------------- async function loadTranslationData() { try { const res = await fetch(CSV_URL); const text = await res.text(); const rows = text.split("\n").slice(1); // Überschriften überspringen translationExamples = []; for (let row of rows) { const cols = row.split("\t"); if (cols.length >= 2 && cols[0].trim()) { translationExamples.push({ de: cols[0].trim(), ez: cols[1].trim() }); } } console.log(`${translationExamples.length} Lern-Beispiele geladen.`); } catch(e) { console.error("Fehler beim Laden der Satz-Datenbank:", e); } } // ------------- KI ÜBERSETZUNG (AI ENGINE) ------------- async function fetchAIWithLearning(userInput) { // 1. Wir bauen den "Lehr-Prompt" aus deinen Tabellendaten let promptInstruction = "Du bist ein professioneller Übersetzer für Jesidisch (Kurmanji). Übersetze den folgenden deutschen Satz präzise in das Jesidische. Nutze die folgenden Beispiele, um den korrekten Satzbau (SOV), den Split-Ergativ in der Vergangenheit und die Ezife-Konstruktion zu lernen. Antworte AUSSCHLIESSLICH mit dem übersetzten Satz, ohne Erklärungen.\n\n### BEISPIELE:\n"; // Die ersten 30-50 Sätze aus deinem Sheet als Lernfutter anhängen translationExamples.slice(0, 50).forEach(ex => { promptInstruction += `Deutsch: ${ex.de} -> Jesidisch: ${ex.ez}\n`; }); promptInstruction += `\n### AUFGABE:\nDeutsch: ${userInput} -> Jesidisch:`; // 2. Anfrage an die KI-Schnittstelle senden try { const response = await fetch("https://api.openai.com/v1/chat/completions", { method: "POST", headers: { "Content-Type": "application/json", "Authorization": `Bearer ${OPENAI_API_KEY}` }, body: JSON.stringify({ model: "gpt-4o-mini", // Kostengünstig, schnell und extrem smart messages: [{ role: "user", content: promptInstruction }], temperature: 0.2 // Niedrige Kreativität = präzisere Grammatik }) }); const data = await response.json(); return data.choices[0].message.content.trim(); } catch (error) { console.error("KI-Schnittstellen-Fehler:", error); return "Übersetzung momentan nicht verfügbar."; } } // ------------- INTERFACE BINDUNG ------------- const searchInput = document.getElementById("searchInput"); let searchTimeout = null; searchInput.addEventListener("input", function() { clearTimeout(searchTimeout); searchTimeout = setTimeout(performAiSearch, 500); // Etwas mehr Zeit beim Tippen lassen }); async function performAiSearch() { const query = searchInput.value.trim(); const banner = document.getElementById("sentence-banner"); const translationOutput = document.getElementById("sentence-translation"); if (!query) { banner.style.display = "none"; return; } // Anzeige auf "lädt..." setzen translationOutput.innerText = "⏳ KI übersetzt und lernt..."; banner.style.display = "block"; // KI-Übersetzung triggern const result = await fetchAIWithLearning(query); // Ergebnis anzeigen translationOutput.innerHTML = `⚡ ${result} (AI Engine v4)`; } // Start der App loadTranslationData();
