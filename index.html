// ------------- CONFIG & STORAGE ------------- let dictionary = {}; let allEzWords = []; // Nutze hier deine Google Sheets TSV-Export-URL const CSV_URL = "https://docs.google.com/spreadsheets/d/1haD4IVA8lNqoUsnrMrnEHhf5iJIGVXGfTWWu5XxevJY/export?format=tsv"; // Zentrales Grammatik-Objekt für Personalendungen im Präsens und Präteritum const GRAMMAR_RULES = { verbs: { "az": { pres_end: "om", past_end: "îm" }, // ich "tu": { pres_end: "î", past_end: "î" }, // du "aw": { pres_end: "e", past_end: "" }, // er/sie/es "em": { pres_end: "în", past_end: "în" }, // wir "hun": { pres_end: "in", past_end: "in" }, // ihr "gîst":{ pres_end: "in", past_end: "in" } // sie (Plural) }, auxiliary_pres: { // Hilfsverb "sein" im Präsens "az": "im", "tu": "î", "aw": "e", "em": "în", "hun": "in", "gîst": "in" }, auxiliary_past: { // Hilfsverb "sein" im Präteritum (war / gewesen) "az": "bûm", "tu": "bûyî", "aw": "bû", "em": "bûn", "hun": "bûn", "gîst": "bûn" } }; // ------------- DICTIONARY LOADER (SPALTEN A BIS R) ------------- async function loadDictionary() { try { const res = await fetch(CSV_URL); const text = await res.text(); const rows = text.split("\n").slice(1); // Überschriftenzeile überspringen dictionary = {}; for (let row of rows) { const cols = row.split("\t"); if (cols.length >= 2 && cols[0].trim()) { const ez = cols[0].trim().toLowerCase(); dictionary[ez] = { de: cols[1] ? cols[1].trim().toLowerCase() : "", type: cols[2] ? cols[2].trim().toLowerCase() : "noun", gender: cols[3] ? cols[3].trim().toLowerCase() : "m", kasus_obliquus: cols[4] ? cols[4].trim() : "", ezife: cols[5] ? cols[5].trim() : "", transitive: cols[6] ? cols[6].trim().toLowerCase() : "intransitiv", v_pres_stem: cols[7] ? cols[7].trim() : "", v_past_stem: cols[8] ? cols[8].trim() : "", v_pres_prefix: cols[9] ? cols[9].trim() : "di", v_cont_prefix: cols[10] ? cols[10].trim() : "di", v_perf_prefix: cols[12] ? cols[12].trim() : "", v_perf_part: cols[13] ? cols[13].trim() : "", pronoun_abs: cols[15] ? cols[15].trim().toLowerCase() : "", pronoun_obl: cols[16] ? cols[16].trim().toLowerCase() : "" }; } } localStorage.setItem("ez_dict_cache_v3", JSON.stringify(dictionary)); } catch(e) { console.warn("Offline-Modus: Nutze lokalen Cache."); dictionary = JSON.parse(localStorage.getItem("ez_dict_cache_v3")) || {}; } allEzWords = Object.keys(dictionary); updateWordOfDay(); initQuiz(); } // ------------- FUZZY MATCHING (FEHLERTOLERANZ) ------------- function getLevenshteinDistance(a, b) { const matrix = []; for (let i = 0; i <= b.length; i++) matrix[i] = [i]; for (let j = 0; j <= a.length; j++) matrix[0][j] = j; for (let i = 1; i <= b.length; i++) { for (let j = 1; j <= a.length; j++) { if (b.charAt(i - 1) === a.charAt(j - 1)) matrix[i][j] = matrix[i - 1][j - 1]; else matrix[i][j] = Math.min(matrix[i-1][j-1]+1, matrix[i][j-1]+1, matrix[i-1][j]+1); } } return matrix[b.length][a.length]; } function findBestFlexMatch(inputWord) { let cleanInput = inputWord.toLowerCase().trim().replace(/[.,\/#!$%\^&\*;:{}=\-_`~()]/g,""); if (!cleanInput) return null; for (let [ez, data] of Object.entries(dictionary)) { if (ez === cleanInput || data.de === cleanInput) return { ez, data, distance: 0 }; } let bestMatch = null; let minDistance = 3; for (let [ez, data] of Object.entries(dictionary)) { let distEz = getLevenshteinDistance(cleanInput, ez); let distDe = getLevenshteinDistance(cleanInput, data.de); let currentMin = Math.min(distEz, distDe); if (currentMin < minDistance) { minDistance = currentMin; bestMatch = { ez, data, distance: currentMin }; } } return bestMatch; } // ------------- DIE ERWEITERTE GRAMMATIK-ENGINE (SOV + ERGATIV) ------------- function translateSmartSentence(inputSentence) { let clean = inputSentence.toLowerCase().replace(/[.,\/#!$%\^&\*;:{}=\-_`~()]/g,""); let words = clean.split(/\s+/); if (words.length === 1) { let sampleMatch = findBestFlexMatch(words[0]); return sampleMatch ? sampleMatch.ez : words[0]; } // Standard-Variablen für Satzstruktur let subjectKey = "aw"; // Standard-Subjekt: er/sie/es let subjectData = null; let objectEz = ""; let objectData = null; let verbData = null; let adjData = null; let adjEz = ""; let conjunctionEz = ""; // Zeitformen & Numerus über deutsche Signalwörter erkennen let isPlural = words.includes("viele") || words.includes("alle") || (words.includes("die") && words.some(w => w.endsWith("e") && w !== "die")); let isIndefinite = words.includes("ein") || words.includes("eine") || words.includes("einen"); let isComparative = words.some(w => w.endsWith("er") && w.length > 3) || words.includes("mehr"); let isSuperlative = words.some(w => w.startsWith("am") || w.endsWith("ste") || w.endsWith("sten")); // Zeitform-Flags let isContinuous = words.includes("gerade") || words.includes("momentan"); let isFutur = words.includes("wird") || words.includes("werden") || words.includes("später"); let isPast = words.includes("war") || words.includes("hatte") || words.includes("hat") || words.includes("schon") || words.some(w => w.startsWith("ge") && w.length > 4); // 1. Wörter anhand der Tabelle klassifizieren words.forEach(word => { let match = findBestFlexMatch(word); if (match) { switch(match.data.type) { case "pronoun": if (match.data.pronoun_abs) { subjectKey = match.data.pronoun_abs; subjectData = match.data; } else { objectEz = match.ez; } break; case "noun": objectEz = match.ez; objectData = match.data; break; case "verb": verbData = match.data; break; case "adjective": adjEz = match.ez; adjData = match.data; break; case "conjunction": conjunctionEz = match.ez; break; } } }); // 2. GRAMMATIK-TRANSFORMATIONEN // A) Nomen anpassen (Kasus, Artikel & Plural) if (objectEz && objectData) { let currentGender = isPlural ? "p" : objectData.gender; let suffix = objectData.kasus_obliquus || (currentGender === "f" ? "ê" : "î"); if (isIndefinite) { objectEz = objectEz + "ek" + suffix; // z.B. bananekê } else if (isPlural) { objectEz = objectEz + "an"; // Pluralendung -an } else if (!isIndefinite && objectData.type === "noun") { objectEz = objectEz + suffix; // Bestimmter Akkusativ } } // B) Adjektiv-Steigerung & Ezife-Konstruktion if (adjEz && adjData && objectData) { if (isComparative) adjEz = adjEz + "tir"; // Komparativ-Suffix if (isSuperlative) adjEz = "herî " + adjEz; // Superlativ-Präfix let glue = objectData.ezife || (objectData.gender === "f" ? "a" : "ê"); if (isPlural) glue = "êt"; objectEz = `${objectEz} ${glue} ${adjEz}`; } // C) Ergativ-Check für das Subjekt (Wenn Vergangenheit UND transitiv) let finalSubject = subjectKey; if (isPast && verbData && verbData.transitive === "transitiv" && subjectData) { finalSubject = subjectData.pronoun_obl || subjectKey; // Wechselt z.B. von "aw" zu "wî" } // D) Verb-Konjugation nach Zeitform let finalVerb = ""; if (verbData) { if (isPast) { if (verbData.v_perf_part) { // Perfekt-Struktur (Partizip + Endung) let end = (verbData.transitive === "transitiv") ? "iye" : (GRAMMAR_RULES.auxiliary_pres[subjectKey] || "e"); finalVerb = verbData.v_perf_part + end; } else { // Präteritum-Struktur let stem = verbData.v_past_stem || verbData.v_pres_stem; let end = (verbData.transitive === "transitiv") ? "" : (GRAMMAR_RULES.verbs[subjectKey]?.past_end || ""); finalVerb = stem + end; } } else if (isFutur) { // Futur-Struktur (Präfix bi- + Präsensstamm) let stem = verbData.v_pres_stem; finalVerb = "bi" + stem + "ê"; if (subjectKey === "aw") finalSubject = "awê"; // Futur-Pronomen-Marker für 3. Person } else if (isContinuous) { // Present Continuous Verlaufsform let prefix = verbData.v_cont_prefix || "di"; let stem = verbData.v_pres_stem; let end = GRAMMAR_RULES.verbs[subjectKey]?.pres_end || "e"; finalVerb = prefix + stem + end; } else { // Standard Präsens let prefix = verbData.v_pres_prefix || "di"; let stem = verbData.v_pres_stem; let end = GRAMMAR_RULES.verbs[subjectKey]?.pres_end || "e"; finalVerb = prefix + stem + end; } } else { // Reine Hilfsverben (sein) ohne Hauptverb if (isPast) { finalVerb = GRAMMAR_RULES.auxiliary_past[subjectKey] || "bû"; } else { finalVerb = GRAMMAR_RULES.auxiliary_pres[subjectKey] || "e"; } } // Adverbien wie "später" temporär abfangen let adverbEz = words.includes("später") ? "piştî re" : ""; // 3. FINALE SOV-ZUSAMMENSETZUNG let finalSentence = ""; if (conjunctionEz) { // Falls ein Bindewort existiert (Zwei Teilsätze) finalSentence = `${finalSubject} ${objectEz} ${finalVerb}`; } else { // Normaler Hauptsatz if (adverbEz) { finalSentence = `${finalSubject} ${adverbEz} ${objectEz} ${finalVerb}`; } else { finalSentence = `${finalSubject} ${objectEz} ${finalVerb}`; } } // Bereinigen und ersten Buchstaben großziehen finalSentence = finalSentence.trim().replace(/\s+/g, " "); return finalSentence.charAt(0).toUpperCase() + finalSentence.slice(1); } // ------------- AB HIER BESTEHENDE USER-INTERFACE LOGIK ------------- const searchInput = document.getElementById("searchInput"); let searchTimeout = null; searchInput.addEventListener("input", function() { clearTimeout(searchTimeout); searchTimeout = setTimeout(performSearch, 300); }); async function performSearch() { const query = searchInput.value.trim(); const resultsDiv = document.getElementById("search-results"); const banner = document.getElementById("sentence-banner"); if (!query) { resultsDiv.innerHTML = ""; banner.style.display = "none"; return; } // Satzbau berechnen const smartTranslation = translateSmartSentence(query); document.getElementById("sentence-translation").innerHTML = `⚡ ${smartTranslation} (Grammatik-Engine v3)`; banner.style.display = "block"; let bestMatch = findBestFlexMatch(query); let html = ""; if (bestMatch) { let ez = bestMatch.ez; let data = bestMatch.data; let note = bestMatch.distance > 0 ? `

`; } resultsDiv.innerHTML = html; } function updateWordOfDay() { if (!allEzWords.length) return; const idx = new Date().getDate() % allEzWords.length; const ez = allEzWords[idx]; if(document.getElementById("daily-ez")) document.getElementById("daily-ez").innerText = ez; if(document.getElementById("daily-de")) document.getElementById("daily-de").innerText = dictionary[ez]?.de || "-"; } function initQuiz() { if (!allEzWords.length || !document.getElementById("quiz-question")) return; const correct = allEzWords[Math.floor(Math.random() * allEzWords.length)]; const correctDe = dictionary[correct].de; let opts = [correctDe]; while(opts.length < 4) { let rand = dictionary[allEzWords[Math.floor(Math.random() * allEzWords.length)]].de; if(!opts.includes(rand)) opts.push(rand); } opts.sort(() => Math.random() - 0.5); document.getElementById("quiz-question").innerHTML = `Was bedeutet ${correct}?`; const optsDiv = document.getElementById("quiz-options"); optsDiv.innerHTML = ""; opts.forEach(opt => { const btn = document.createElement("button"); btn.className = "quiz-option"; btn.innerText = opt; btn.onclick = () => { const isCorrect = (opt === correctDe); btn.classList.add(isCorrect ?
